on_actions = {
	
	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					HUN = { has_cosmetic_tag = HUN_EMPIRE }
					country_exists = AUS
				}
				HUN = {
					drop_cosmetic_tag = yes
				}
			}
			if = {
				limit = {
					AUS = { has_cosmetic_tag = HUN_EMPIRE }
					country_exists = HUN
				}
				AUS = {
					drop_cosmetic_tag = yes
				}
			}
			if = {
				limit = {
					CZE = { has_cosmetic_tag = HUN_EMPIRE }
					OR = {
						country_exists = AUS
						country_exists = HUN
					}
				}
				CZE = {
					drop_cosmetic_tag = yes
				}
			}
		}
	}
	
	on_startup = {
			effect = {
				VEN = {
					#give_resource_rights = { receiver = ENG state = 476 }
					#give_resource_rights = { receiver = USA state = 480 }
				}
				# Game Rules - nano countries
				if = {
					limit = {
						has_game_rule = {
							rule = nano_nation
							option = SFS_BLOCKED
						}
					}
					SPR = {
						annex_country = {
							target = ADR
							transfer_troops = no
						}
					}
					FRA = {
						annex_country = {
							target = MCO
							transfer_troops = no
						}
					}
					ITA = {
						annex_country = {
							target = SMR
							transfer_troops = no
						}
					}
					SWI = {
						annex_country = {
							target = LIE
							transfer_troops = no
						}
					}
				}
				# Game Rules - Czechoslovakia fragmentation - CZE-SLO
				if = {
					limit = {
						has_game_rule = {
							rule = cze_fragmentation_status
							option = CZE_SLO_FRAGMENTED
						}
					}
					CZE = {
						release = SLO
						set_cosmetic_tag = CZE_ONLY
					}
				}
				# Game Rules - Czechoslovakia fragmentation - Fully
				if = {
					limit = {
						has_game_rule = {
							rule = cze_fragmentation_status
							option = CZE_FRAGMENTED
						}
					}
					CZE = {
						release = SLO
						release = MRV
						set_cosmetic_tag = BOHEMIA
					}
				}
			}
	}
	
	on_puppet = {
		effect = {
			if = {
				limit = {
					AND = {
						TAG = HUN
						is_subject_of = GER
						GER = { has_government = fascism }
					}
				}
				set_cosmetic_tag = HUN_GER
				set_party_name = { ideology = fascism long_name = HUN_Arrow_Cross_party_long name = HUN_Arrow_Cross_party }
				#create_country_leader = {
				#	name = "Ferenc Szálasi"
				#	desc = "POLITICS_FERENC_SZALASI_DESC"
				#	picture = "portrait_hun_ferenc_szalasi.dds"
				#	expire = "1965.1.1"
				#	ideology = fascism_ideology
				#	traits = {
				#	}
				#}
			}
		
		}
	}
	
	on_release_as_puppet = {
		effect = {
			if = {
				limit = {
					AND = {
						TAG = HUN
						is_subject_of = GER
						GER = { has_government = fascism }
					}
				}
				set_cosmetic_tag = HUN_GER
				set_party_name = { ideology = fascism long_name = HUN_Arrow_Cross_party_long name = HUN_Arrow_Cross_party }
				#create_country_leader = {
				#	name = "Ferenc Szálasi"
				#	desc = "POLITICS_FERENC_SZALASI_DESC"
				#	picture = "portrait_hun_ferenc_szalasi.dds"
				#	expire = "1965.1.1"
				#	ideology = fascism_ideology
				#	traits = {
				#	}
				#}
			}
		}
	}
	
	on_release_as_free = {
		effect = {
			if = {
				limit = {
					OR = {
						original_tag = MAL
						original_tag = BRM
					}
				}
				abandon_colony_tag = yes
			}
		}
	}
	
	on_government_change = {
		effect = {
			if = {
				limit = {
					has_dlc = "Death or Dishonor"
					has_cosmetic_tag = HUN_EMPIRE
					has_government = fascism
				}
				country_event = { id = AUH.1 }
			}
			if = {
				limit = {
					has_dlc = "Death or Dishonor"
					has_cosmetic_tag = HUN_EMPIRE
					has_government = communism
				}
				country_event = { id = AUH.2 }
			}
		}
	}
	
	on_declare_war = {
		effect = {
			if = {
				limit = {
					OR = {
						original_tag = MAL
						original_tag = BRM
					}
				}
				if = {
					limit = {
						ROOT = {
							original_tag = FROM
						}
					}
					sfs_abandon_colony_tag = yes
					FROM = {
						sfs_abandon_colony_tag = yes
					}
				}
			}
		}
	}
	
	on_annex = {
		effect = {
			every_country = {
				limit = {
					OR = {
						original_tag = MAL
						original_tag = BRM
					}
					is_puppet = no
				}
				sfs_abandon_colony_tag = yes
			}
		}
	}
	
	on_subject_free = {
		effect = {
			#make sure Dominions lose the dominion tag
			if = {
				limit = {
					OR = {
						original_tag = MAL
						original_tag = BRM
						original_tag = SAW
					}
				}
				abandon_colony_tag = yes
			}

			if = {
				limit = {
					TAG = MAL
				}
				retire_country_leader = yes
			}
			
			if = {
				limit = {
					TAG = SAW
				}
				retire_country_leader = yes
			}
			
			if = {
				limit = {
					TAG = BRM
				}
				retire_country_leader = yes
			}
		}
	}
}
